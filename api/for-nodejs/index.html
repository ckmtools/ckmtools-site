Node.js patterns

fetch — built-in (Node 18+, Deno, Bun) recommended 
const TEXTLENS_KEY = process.env.TEXTLENS_KEY;

async function analyze(text) {
  const res = await fetch('https://api.ckmtools.dev/v1/analyze', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': TEXTLENS_KEY,
    },
    body: JSON.stringify({ text }),
  });
  if (!res.ok) throw new Error(`TextLens error: ${res.status}`);
  return res.json();
}

const result = await analyze(article.body);
console.log(`Grade: ${result.readability.consensusGrade}`);
console.log(`Sentiment: ${result.sentiment.label}`);
console.log(`SEO score: ${result.seo.score}/100`);
Promise.all — batch analysis 
async function analyzeBatch(texts) {
  return Promise.all(
    texts.map(text =>
      fetch('https://api.ckmtools.dev/v1/analyze', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-API-Key': process.env.TEXTLENS_KEY,
        },
        body: JSON.stringify({ text }),
      }).then(r => r.json())
    )
  );
}

const results = await analyzeBatch(articles.map(a => a.body));
results.forEach((r, i) => {
  console.log(`${articles[i].title}: grade ${r.readability.consensusGrade}`);
});
Express middleware — auto-score on save 
import express from 'express';

const app = express();
app.use(express.json());

app.post('/articles', async (req, res) => {
  const { title, body } = req.body;

  const analysis = await fetch('https://api.ckmtools.dev/v1/analyze', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': process.env.TEXTLENS_KEY,
    },
    body: JSON.stringify({ text: body }),
  }).then(r => r.json());

  const article = await db.articles.create({
    title,
    body,
    readability_grade: analysis.readability.consensusGrade,
    sentiment_label: analysis.sentiment.label,
    seo_score: analysis.seo.score,
    keywords: analysis.keywords.slice(0, 5).map(k => k.word),
  });

  res.json(article);
});

Why not textstat, VADER, and yake?

Those are Python libraries — you'd need a Python sidecar process just to call them from Node.js. There's no native JS equivalent that covers all three: readability, sentiment, and keyword extraction in a single package. TextLens API handles all of it in one HTTP call. Works in Node.js, Deno, Bun, Cloudflare Workers, AWS Lambda, or any runtime that can make a POST request. No Python runtime, no subprocess, no separate service to maintain.
